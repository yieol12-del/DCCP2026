Q1 (1점) - 학생 성적 보고서 생성기
(1) 코드
Python
import csv
import json
import logging

# [cite: 19] 로깅 설정 (level=logging.INFO)
logging.basicConfig(level=logging.INFO, format='%(levelname)s: %(message)s')

def make_report(csv_path: str, json_path: str) -> int:
    students_data: list[dict] = []
    success_count: int = 0

    try:
        # [cite: 3, 11, 12, 14] 파일 읽기 (utf-8 명시, DictReader 사용)
        with open(csv_path, "r", encoding="utf-8") as f:
            reader = csv.DictReader(f)
            for row in reader:
                try:
                    name: str = row["이름"]
                    student_id: str = row["학번"]
                    
                    # [cite: 13, 15] 결측값 확인 및 점수 변환
                    scores_raw = [row["중간"], row["기말"], row["과제"]]
                    
                    if "" in scores_raw:
                        # [cite: 15] 빈 칸이 있는 경우 None 처리
                        scores_dict = {
                            "중간": int(row["중간"]) if row["중간"] else None,
                            "기말": int(row["기말"]) if row["기말"] else None,
                            "과제": int(row["과제"]) if row["과제"] else None
                        }
                        avg, grade = None, None
                        logging.info(f"{name}: 결측값이 포함되어 성적 산출 불가") [cite: 21]
                    else:
                        # [cite: 16, 17] 가중 평균 및 등급 계산
                        mid, final, assign = map(int, scores_raw)
                        scores_dict = {"중간": mid, "기말": final, "과제": assign}
                        avg = mid * 0.3 + final * 0.5 + assign * 0.2
                        
                        if avg >= 90: grade = "A"
                        elif avg >= 80: grade = "B"
                        elif avg >= 70: grade = "C"
                        else: grade = "F"
                        logging.info(f"{name}: 평균 {avg:.1f}, 등급 {grade}") [cite: 21]
                    
                    #  JSON 저장을 위한 구조 생성
                    students_data.append({
                        "이름": name,
                        "학번": student_id,
                        "점수": scores_dict,
                        "평균": avg,
                        "등급": grade
                    })
                    success_count += 1
                    
                except ValueError:
                    logging.error(f"{row.get('이름', '알 수 없음')} 학생의 점수 형식이 잘못되었습니다.")
                    continue

        # [cite: 17] 결과를 JSON으로 저장 (한글 보존, 들여쓰기 적용)
        with open(json_path, "w", encoding="utf-8") as f:
            json.dump(students_data, f, ensure_ascii=False, indent=4)
            
        return success_count

    except FileNotFoundError:
        # [cite: 19] 파일이 없는 경우
        logging.warning(f"CSV 파일이 존재하지 않습니다: {csv_path}")
        return 0
    except UnicodeDecodeError:
        # [cite: 20] 인코딩이 잘못된 경우
        logging.error(f"인코딩 오류: {csv_path}의 인코딩을 확인하세요 (UTF-8 필요).")
        return 0

Q2 (1점) - 사용자 정의 예외와 자모 분류
(1) 코드
Python
# (a) 사용자 정의 예외 클래스 정의
class InvalidJamoError(ValueError):
    pass

# (b) 자모 분류 함수 구현
def classify_jamo(c: str) -> str:
    # 1. 타입 검사
    if not isinstance(c, str):
        raise TypeError("입력값은 문자열(str)이어야 합니다.")
    
    # 2. 길이 검사 (InvalidJamoError가 아닌 일반 ValueError)
    if len(c) != 1:
        raise ValueError("입력된 문자열의 길이는 1이어야 합니다.")
    
    code = ord(c)
    
    # 3. 유니코드 범위 검사
    # 한글 자음 (ㄱ-ㅎ)
    if 0x3131 <= code <= 0x314E:
        return "자음"
    # 한글 모음 (ㅏ-ㅣ)
    elif 0x314F <= code <= 0x3163:
        return "모음"
    # 그 외 한 글자
    else:
        raise InvalidJamoError(f"'{c}'은(는) 유효한 한글 자모가 아닙니다.")

# (c) 리스트 순회 및 예외 처리
inputs = ["ㄱ", "ㅏ", "가", " ", "AB", 5, "ㅎ", "ㅣ"]

for item in inputs:
    try:
        result = classify_jamo(item)
        print(f"'{item}' -> {result}")
    
    # 예외 처리: 자식 예외인 InvalidJamoError를 먼저 잡아야 함
    except TypeError as e:
        print(f"[TypeError] {e}")
    except InvalidJamoError as e:
        print(f"[InvalidJamoError] {e}")
    except ValueError as e:
        print(f"[ValueError] {e}")

(3) 설명
Exception이 아닌 ValueError를 상속받은 이유: 입력받은 인자가 올바른 타입(문자열)이지만, 그 '값(Value)'이 자모 분류라는 목적에 부합하지 않기 때문입니다. 최상위 예외인 Exception을 직접 상속받으면 다른 치명적인 시스템 오류들과 구분하기 어려워지므로, 성격에 맞는 내장 예외(ValueError)의 자식으로 만들어 예외의 의미를 명확히 하는 것이 파이썬의 권장 스타일(Pythonic)입니다.

예외 처리 순서: InvalidJamoError가 ValueError를 상속받았기 때문에, except 블록을 작성할 때 반드시 자식 클래스인 InvalidJamoError를 먼저 잡아주어야 합니다. 만약 부모인 ValueError를 먼저 적으면, 다형성에 의해 InvalidJamoError도 부모 쪽으로 빨려 들어가서 잘못된 메시지가 출력됩니다.

Q1에서 코드가 뭘 해주는 코드야

작성해 드린 Q1 코드는 쉽게 말해 "자동 성적 산출 및 보고서 생성 봇" 역할을 합니다. 흩어져 있는 원시 데이터(CSV)를 가공해서 보기 좋은 결과물(JSON)로 만들어주는 과정을 거칩니다.

코드의 흐름을 4가지 핵심 역할로 나누어 설명해 드릴게요.

1. 데이터 불러오기 및 검사 (CSV 읽기)
scores.csv 파일을 열어 학생들의 이름, 학번, 각 과목 점수를 한 줄씩 읽어옵니다.

이때 점수 칸이 하나라도 비어있는 학생을 찾아내어, 억지로 계산하지 않고 평균과 등급을 None으로 안전하게 비워둡니다.

2. 성적 산출 (가중 평균 및 등급 계산)
점수가 모두 정상적으로 입력된 학생들에 한해, 정해진 비율(중간 30%, 기말 50%, 과제 20%)을 반영하여 최종 평균 점수를 계산합니다.

계산된 평균을 기준으로 90점 이상은 A, 80점 이상은 B 등 정해진 기준표에 따라 등급을 부여합니다.

3. 결과물 저장 (JSON 쓰기)
모든 계산이 끝나면 학생들의 성적표를 모아 report.json이라는 새로운 파일로 예쁘게 포장해서 저장합니다.

단순히 데이터를 저장하는 것을 넘어, 한글이 깨지지 않게 보호하고 줄바꿈과 들여쓰기 공간을 주어 사람이 문서를 열었을 때 한눈에 읽기 편하도록 돕습니다.

4. 안전장치 및 실시간 기록 (예외 처리와 로깅)
작업이 진행되는 동안 화면에 "누구의 평균은 몇 점이다"라는 식으로 진행 상황을 실시간으로 중계(Logging)합니다.

만약 읽어야 할 파일이 아예 없거나(FileNotFoundError), 파일의 인코딩이 잘못되어 글자가 깨져있다면(UnicodeDecodeError), 프로그램이 갑자기 뻗어버리는 대신 "파일을 확인해 주세요"라는 경고 메시지를 남기고 부드럽게 대처하도록 방어막(try-except)을 쳐두었습니다.

결국 이 코드는 실무 데이터 분석이나 서버 개발에서 가장 기본이 되는 ETL(Extract-데이터 추출, Transform-데이터 변환, Load-데이터 저장) 과정을 파이썬으로 작게 구현해 놓은 축소판이라고 보시면 됩니다.

import csv
import json
import logging

logging.basicConfig(level=logging.INFO)

def make_report(csv_path: str, json_path: str) -> int:
    success_count: int = 0
    results: list[dict] = []
    
    try:
        with open(csv_path, "r", encoding="utf-8") as f:
            reader = csv.DictReader(f)
            for row in reader:
                # CSV 헤더나 값에 있을 수 있는 공백 제거
                clean_row: dict[str, str] = {k.strip(): v.strip() for k, v in row.items()}
                
                name: str = clean_row["이름"]
                student_id: str = clean_row["학번"]
                m_str: str = clean_row["중간"]
                f_str: str = clean_row["기말"]
                a_str: str = clean_row["과제"]
                
                # 결측값(빈 문자열) 확인
                if m_str == "" or f_str == "" or a_str == "":
                    scores: dict[str, int | None] = {
                        "중간": int(m_str) if m_str != "" else None,
                        "기말": int(f_str) if f_str != "" else None,
                        "과제": int(a_str) if a_str != "" else None
                    }
                    avg = None
                    grade = None
                    logging.info(f"{name}: 결측값이 있어 성적을 산출할 수 없습니다.")
                else:
                    mid: int = int(m_str)
                    fin: int = int(f_str)
                    asn: int = int(a_str)
                    scores: dict[str, int | None] = {"중간": mid, "기말": fin, "과제": asn}
                    
                    avg: float = mid * 0.3 + fin * 0.5 + asn * 0.2
                    
                    if avg >= 90:
                        grade = "A"
                    elif avg >= 80:
                        grade = "B"
                    elif avg >= 70:
                        grade = "C"
                    else:
                        grade = "F"
                    
                    logging.info(f"{name}: 평균 {avg}, 등급 {grade}")
                
                results.append({
                    "이름": name,
                    "학번": student_id,
                    "점수": scores,
                    "평균": avg,
                    "등급": grade
                })
                success_count += 1
                
    except FileNotFoundError:
        logging.warning(f"파일이 존재하지 않습니다: {csv_path}")
        return 0
    except UnicodeDecodeError:
        logging.error(f"인코딩이 잘못되었습니다: {csv_path}")
        return 0
        
    # JSON 파일 쓰기
    with open(json_path, "w", encoding="utf-8") as f:
        json.dump(results, f, ensure_ascii=False, indent=4)
        
    return success_count

(3) 설명. * 결측값 처리 전략: csv.DictReader로 데이터를 읽을 때, 점수 값이 들어갈 자리가 빈 문자열("")인지 조건문으로 확인했습니다. 빈 문자열이 하나라도 발견되면 평균과 등급 변수에 파이썬의 None을 할당하여, json.dump 수행 시 자동으로 JSON 표준인 null로 변환되도록 처리했습니다.

인코딩 선택의 근거: 문제에 제시된 원본 데이터(scores.csv)가 한글을 포함하고 있으므로, 파일 입출력 과정에서 데이터가 깨지지 않도록 한글 텍스트 처리의 표준이자 가장 범용적인 utf-8을 명시했습니다. 추가로 JSON 저장 시 ensure_ascii=False를 설정하여 유니코드 이스케이프 문자(\uXXXX)가 아닌 사람이 읽기 쉬운 한글 원형 그대로 보존되도록 작성했습니다.
