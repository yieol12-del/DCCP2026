# (a) search_word 함수 작성
def search_word(q: str, num: int = 10, start: int = 1) -> dict:
    url = "https://opendict.korean.go.kr/api/search"
    params = {
        "key": KEY,
        "q": q,
        "req_type": "json",
        "num": num,
        "start": start,
        "type1": "word" # PDF의 typel은 type1의 오타이므로 type1으로 작성
    }
    r = requests.get(url, params=params, timeout=10)
    r.raise_for_status() # HTTP 에러 발생 시 예외 처리
    return r.json()

# (b) 응답 구조 살펴보기
data = search_word("김치")
print("--- JSON 응답 구조 ---")
print(json.dumps(data, ensure_ascii=False, indent=2)[:400])
print("\n" + "="*50 + "\n")

# (c) 원하는 정보만 추출
channel = data.get("channel", {})
total = channel.get("total", 0)
items = channel.get("item", [])
n = len(items)

# (i) 전체 결과 수와 이 페이지 항목 수 출력
print(f"총 {total}건, 이 페이지 {n}건\n")

# (ii) 첫 5개 항목 추출 및 출력
for item in items[:5]:
    word = item.get("word", "")
    pos = item.get("pos", "품사 없음")
    
    # API 구조상 sense 필드 안에 definition(뜻풀이)이 존재함
    sense = item.get("sense", {})
    # sense가 리스트로 오는 경우와 딕셔너리로 오는 경우 모두 안전하게 처리
    if isinstance(sense, list):
        definition = sense[0].get("definition", "")
    else:
        definition = sense.get("definition", "")
        
    print(f"{word} ({pos}) -> {definition[:40]}")

import time
from collections import Counter

# 검색할 음식 단어 리스트 [cite: 123, 124, 125, 126]
words: list[str] = [
    "김치", "라면", "만두", "김밥", 
    "국수", "떡볶이", "불고기", "비빔밥"
]

all_items: list[dict] = []

print("--- (a) 검색어별 결과 수 ---")
for word in words:
    # Q1에서 만든 함수 호출 [cite: 122]
    data = search_word(word)
    
    # 전체 결과 수 추출 및 출력 [cite: 128, 129]
    total: int = data.get("channel", {}).get("total", 0)
    print(f"{word}: {total}건")
    
    # 가져온 10개의 항목을 전체 리스트에 병합 
    items: list[dict] = data.get("channel", {}).get("item", [])
    all_items.extend(items)
    
    # 서버에 부담을 주지 않기 위해 0.3초 대기 
    time.sleep(0.3)

print("\n--- (b) 품사 빈도 상위 3개 ---")
pos_list: list[str] = []

for item in all_items:
    # pos 필드가 없거나 빈 문자열("")인 경우 "(미상)"으로 처리 
    pos: str = item.get("pos") or "(미상)"
    pos_list.append(pos)

# 빈도 계산 및 상위 3개 추출 [cite: 133, 136]
counter = Counter(pos_list)
for pos, count in counter.most_common(3):
    print(f"{pos}: {count}회")

.extend() 와 .append()의 차이:

all_items라는 큰 장바구니에 데이터를 담을 때, 만약 all_items.append(items)를 쓰면 장바구니 안에 '리스트 덩어리'들이 들어가서 2차원 리스트([ [], [], [] ])가 되어버립니다.

반면 .extend(items)를 쓰면 덩어리를 풀어서 내용물(딕셔너리)만 알맹이 채로 큰 장바구니에 쏙쏙 담아줍니다. 그래서 총 80개의 아이템이 깔끔한 1차원 리스트로 모이게 됩니다.

🔍 왜 빈칸 ()으로 나올까요?
국어사전에서 김치나 라면 같은 단어들은 단일어이기 때문에 명확하게 '명사'라는 품사가 등록되어 있습니다.

하지만 김치 공장이나 김치 보릿고개 같은 단어들은 국어학적으로 '구(Phrase)' 또는 두 명사가 합쳐진 '복합어'로 분류됩니다. 우리말샘 사전은 개방형 사전이기 때문에 이러한 복합어나 전문 용어들도 대거 등록되어 있는데, 내부 데이터를 정밀하게 살펴보면 이 구나 복합어 항목들은 pos(품사) 필드 자체가 아예 빈 문자열("")로 저장되어 있는 경우가 많습니다.

즉, 데이터가 "pos": "" 상태로 들어오다 보니, 우리가 짠 코드에서 pos = sense[0].get("pos", "품사 없음")를 돌렸을 때:

"pos"라는 키(Key) 자체는 존재하니까 기본값인 "품사 없음"으로 넘어가지 않고,

그 안에 들어있는 알맹이인 빈 문자열("")을 그대로 가져오게 됩니다.

그래서 출력할 때 ({pos})가 ()가 되어버린 것이죠!

🛠️ 더 완벽하게 예외 처리하는 방법
교수님이 보시기에 괄호 안이 비어있는 () 보다는, 품사가 없을 때 (품사 없음) 혹은 (미상)이라고 명확하게 찍히는 게 훨씬 완성도 높은 코드처럼 보입니다.

Q1 코드의 for item in items[:5]: 내부를 아래처럼 or "품사 없음"을 붙여서 아주 살짝만 수정해 보세요. 파이썬에서 빈 문자열("")은 거짓(False) 취급을 받기 때문에, 빈칸일 때 자동으로 뒤에 있는 "품사 없음"이 채택됩니다.

Python
for item in items[:5]:
    word = item.get("word", "")
    sense = item.get("sense", {})
    
    if isinstance(sense, list) and len(sense) > 0:
        # ⚠️ 뒤에 or "품사 없음"을 붙여서 빈 문자열("")일 때를 방어합니다!
        pos = sense[0].get("pos") or "품사 없음"
        definition = sense[0].get("definition", "")
    elif isinstance(sense, dict):
        pos = sense.get("pos") or "품사 없음"
        definition = sense.get("definition", "")
    else:
        pos = "품사 없음"
        definition = ""
        
    print(f"{word} ({pos}) -> {definition[:40]}")
